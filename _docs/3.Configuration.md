# Configuration

## Storage path

```php
<?php

return [
    /* ------------------------------------------------------------------------------------------------
     |  Log files storage path
     | ------------------------------------------------------------------------------------------------
     */
    'storage-path'  => storage_path('logs'),

    // ...
];
```

## Localization

```php
<?php

return [
    // ...

    /* ------------------------------------------------------------------------------------------------
     |  Locale
     | ------------------------------------------------------------------------------------------------
     |  Supported locales :
     |    'auto', 'ar', 'de', 'en', 'es', 'fa', 'fr', 'hy', 'it', 'nl', 'pl', 'pt-BR', 'ro', 'ru', 'sv', 'tr', 'zh-TW', 'zh'
     */
    'locale'        => 'auto',

    // ...
];
```

## Facade

```php
<?php

return [
    // ...

    /* -------------------------------------------------------------------------
     |  LogViewer's Facade
     | -------------------------------------------------------------------------
     */
    'facade'        => 'LogViewer',

    // ...
];
```

## Route

```php
<?php

return [
    // ...

    /* -------------------------------------------------------------------------
     |  Route
     | -------------------------------------------------------------------------
     */
    'route'         => [
        'enabled'    => true,

        'attributes' => [
            'prefix'     => 'log-viewer',

            'middleware' => null,
        ],
    ],

    // ...
];
```

## Pagination

```php
<?php

return [
    // ...

    /* -------------------------------------------------------------------------
     |  Log entries per page
     | -------------------------------------------------------------------------
     |  This defines how many log entries are displayed per page.
     */
    'per-page'      => 30,

    // ...
];
```

## Download

<?php

return [
